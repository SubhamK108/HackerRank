import string
def print_rangoli(s):
    a = string.ascii_lowercase
    l = []
    for i in range(0, s):
        b = '-'.join(a[i:s])
        l.append((b[::-1] + b[1:]).center(4*s - 3, '-'))
    print('\n'.join(l[::-1]))
    print('\n'.join(l[1:]))
if __name__ == '__main__':
    n = int(input())
    print_rangoli(n)
